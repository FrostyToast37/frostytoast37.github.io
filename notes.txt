socket.on("send_private_msg", async (data) => {
    // 1. Data usually looks like: { to: "RecipientUser", content: "Hello!" }
    const { to, content } = data;
    const from = socket.request.session.user.username;

    try {
        // 2. Persistent Storage (The "Shoebox")
        await saveMessage(from, to, content);

        // 3. The Delivery
        // We emit to the receiver's room, and usually back to the sender 
        // so their UI can update to show the message was sent successfully.
        const payload = {
            from,
            content,
            timestamp: new Date()
        };

        io.to(to).to(from).emit("receive_private_msg", payload);

    } catch (err) {
        console.error("Message failed to send:", err);
        socket.emit("error_msg", "Could not deliver message.");
    }
});
